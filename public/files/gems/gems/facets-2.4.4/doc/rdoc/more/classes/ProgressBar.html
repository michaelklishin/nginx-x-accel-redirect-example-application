  def initialize(title, total, out = STDERR)
    @title = title
    @total = total
    @out = out
    @bar_length = 80
    @bar_mark = "o"
    @total_overflow = true
    @current = 0
    @previous = 0
    @is_finished = false
    @start_time = Time.now
    @format = "%-14s %3d%% %s %s"
    @format_arguments = [:title, :percentage, :bar, :stat]
    show_progress
  end

  def set(count)
    if count < 0
      raise "invalid count less than zero: #{count}"
    elsif count > @total
      if @total_overflow
        @total = count + 1
      else
        raise "invalid count greater than total: #{count}"
      end
    end
    @current = count
    show_progress
    @previous = @current
  end
