# File lib/more/facets/fileshell.rb, line 48
  def [](name)
    if File.directory?(name)
      Dir.new(name)
    elsif File.file?(name)
      File.new(name)
    else
      nil
    end
  end

root?(dir=nil)

Is a directory root?

[ show source ]

# File lib/more/facets/fileshell.rb, line 59
  def root?(dir=nil)
    pth = File.expand_path(dir||work)
    return true if pth == '/'
    return true if pth =~ /^(\w:)?\/$/
    false
  end
